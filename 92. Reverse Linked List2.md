class Solution:
    def reverseBetween(self, head: Optional[ListNode], left: int, right: int) -> Optional[ListNode]:

        # Set Two pointer
        dummy = ListNode(0, head)
        prev, cur = dummy, head
        
        # Mv two pointers to left position
        for i in range(left - 1):
            prev, cur = cur, cur.next

        # reverse from left to right
        prev2 = None
        for i in range(right - left + 1):
            nxt = cur.next
            cur.next = prev2
            prev2 = cur
            cur = nxt
        # update pointers
        prev.next.next = cur
        prev.next = prev2
        
        # return head node
        return dummy.next
~~~
