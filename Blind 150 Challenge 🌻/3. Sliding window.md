#### Best Time to Buy and Sell Stock(121)

~~~python
class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        """Q: Find Maximum profit 
        - Set Two pointer(buy, sell) to skim
        - If buy price > sell price
        - Restore Max profit        
        Time: O(n),  Space: O(1)"""
        l, r = 0, 1
        res = 0
        while r < len(prices):
            # Found new lowest price üòô
            if prices[l] >= prices[r]:
                l = r 
            else:
                profit = prices[r] - prices[l]
                res = max(res, profit)
            r += 1
        return res
~~~

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        # "abcabcbb"
        #    l r
        # {c, a, b}
        charSet = set()
        l = 0
        res = 0
        for r in range(len(s)):
            while s[r] in charSet:  # Ïù¥ Î∂ÄÎ∂ÑÏù¥ ÌïµÏã¨ üê≥
                charSet.remove(s[l])
                l += 1
            charSet.add(s[r])
            res = max(r-l+1, res)
        return res

# Time: O(n): Single pass
# Space: O(1)
class Solution:
    def maxArea(self, height: List[int]) -> int:
        # [1,8,6,2,5,4,8,3,7]
        res = 0
        l, r = 0, len(height)-1
        while l < r:
            area = (r - l) * min(height[l], height[r])
            res = max(res, area)
            if height[l] > height[r]:
                r -= 1
            else:
                l += 1
        return res

4. Longest Repeating Character Replacement(424)

~~~python
# idea: k >= (windowLength - count[mostFreq]). windowLengthÍ∞Ä ÏµúÎåÄÍ∞íÏúºÎ°ú Í∞±Ïã†ÎêòÎ†§Î©¥ Ìè¨Ïù∏ÌÑ∞Ïùò Í∞±Ïã†ÏùÄ count[mostFreq]Í∞Ä Ï¶ùÍ∞ÄÌïòÎäî Î∞©Ìñ•Ïù¥Ïñ¥Ïïº Ìö®Ïú®Ï†ÅÏù¥Îã§.

class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        '''
        "AABABBA", k = 1 => 4
           l
              r
         count = {'A':1, 'B':3}
         res == r - l + 1 = 4// if mostFreq + k > res ==> l += 1
        '''
        count = {}
        res = 0
        # Sliding window
        l = 0
        for r in range(len(s)):
            count[s[r]] = count.get(s[r], 0) + 1
            while (r - l + 1) - max(count.values()) > k: # Ïù¥ Î∂ÄÎ∂ÑÏù¥ ÌïµÏã¨ üå∏
                count[s[l]] -= 1
                l += 1
            res = max(res, r - l + 1)    
        return res
~~~
