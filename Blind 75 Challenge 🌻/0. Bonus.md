class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:
        """
        ì „ì œ: ì˜¤ë¦„ì°¨ìˆœ
           [2,7,11,15], 9

        r:      ^
        cusSum = 2 + 11 = 9 > target => r-1

        """
        l, r = 0, len(numbers)-1
        
        while l<r:
            cur_sum = numbers[l] + numbers[r]
            
            if cur_sum > target:
                r -= 1
            elif cur_sum < target:
                l += 1
            else:
                return [l + 1, r + 1]
        return

3. ##### Maximum Difference Between Increasing Elements(2016)

~~~python
# ì£¼ì‹ìœ¼ë¡œ ìˆ˜ìµë‚´ê¸°ì™€ ë§¤ìš° ë¹„ìŠ·í•œ ë¬¸ì œ ê°™ë‹¤.
# ë‘ ìš”ì†Œì˜ ì°¨ê°€ ì¦ê°€ë¶„ì´ ì•„ë‹ˆë¼ë©´ -1ì„ ë°˜í™˜í•˜ë¼
# idea: Slid window
# ë°°ìš¸ ê²ƒ: ê³ ì • ê°’(ìž‘ì€ ê°’)ë³´ë‹¤ ìž‘ì€ ê°’ì´ ë‚˜ì˜¤ë©´ ê·¸ ìœ„ì¹˜ìœ¼ë¡œ ì´ë™í•œë‹¤.
# [3,5,2,10] => 8
# [9,4,3,2] => -1
class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        l = 0
         
        res = 0
        for r in range(1, len(nums)):
            if nums[l] > nums[r]:
                l = r
            res = max(res, nums[r] - nums[l])
        return res if res != 0 else -1    
~~~
